类定义

@<class-name> {
    property: Type
    method(param: Type) {
        # method body
        this.property = ... # 类方法中可使用 this 获取当前示例
    }
}

匿名类：

variable = ${
    property: Type
    method(param: Type) {
        # method body
    }
}

类实例化：

instance = new <class-name>(param)
instance = new <class-name> { property: <value> }

使用示例：

@Point {
    x: Number
    y: Number
    distanceFromOrigin() {
        return math.sqrt(x ^ 2 + y ^ 2)
    }
}

point1 = new Point(1, 1)
point2 = new Point{ x: 1, y: 1 }
